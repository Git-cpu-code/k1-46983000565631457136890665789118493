const overlay = document.getElementById('overlay'); const noteDisplay = document.getElementById('note-display'); const overlayOpacity = 0.9; const activeKeys = new Set(); // Set für aktive Tasten let graceTimeout; const gracePeriod = 50; // Grace Period für "gleichzeitiges Drücken" // Zuordnung von Tasten zu den Audiodateien const sounds = { 'key-a0': 'sounds/a0.mp3', 'key-as0': 'sounds/as0.mp3', 'key-h0': 'sounds/h0.mp3', 'key-c1': 'sounds/c1.mp3', 'key-cs1': 'sounds/cs1.mp3', 'key-d1': 'sounds/d1.mp3', 'key-ds1': 'sounds/ds1.mp3', 'key-e1': 'sounds/e1.mp3', 'key-f1': 'sounds/f1.mp3', 'key-fs1': 'sounds/fs1.mp3', 'key-g1': 'sounds/g1.mp3', 'key-gs1': 'sounds/gs1.mp3', 'key-a1': 'sounds/a1.mp3', 'key-as1': 'sounds/as1.mp3', 'key-h1': 'sounds/h1.mp3' }; const noteDurations = { whole: 4000, half: 2000, quarter: 1000, eighth: 500, sixteenth: 250, thirtysecond: 125 }; const noteImages = { whole: 'note1.png', half: 'note2.png', quarter: 'note3.png', eighth: 'note4.png', sixteenth: 'note5.png', thirtysecond: 'note6.png' }; // Funktion zur Berechnung der vertikalen Position basierend auf der Oktave function calculateNotePosition(octave) { const maxTop = 150; // Höchste Position für die höchste Oktave const minTop = 630; // Niedrigste Position für die tiefste Oktave const topPosition = minTop - (octave * (minTop - maxTop) / 88); return topPosition; } document.querySelectorAll('.white-key,.black-key').forEach(key => { let startTime, endTime; // Gemeinsame Funktion für touchstart/mousedown function handleKeyDown() { clearTimeout(graceTimeout); activeKeys.add(key.id); updateOverlayColor(); startTime = new Date().getTime(); const audio = new Audio(sounds[key.id]); audio.play().catch(error => { const errorMessageDiv = document.getElementById('error-message'); errorMessageDiv.innerText = "Fehler beim Abspielen des Tons: " + error.message; errorMessageDiv.style.display = 'block'; }); } // Gemeinsame Funktion für touchend/mouseup function handleKeyUp() { endTime = new Date().getTime(); handleNoteDisplay(key, endTime - startTime); graceTimeout = setTimeout(() => { activeKeys.delete(key.id); if (activeKeys.size === 0) { overlay.style.display = 'none'; } else { updateOverlayColor(); } }, gracePeriod); } // Event Listener für Touch-Geräte key.addEventListener('touchstart', (e) => { e.preventDefault(); handleKeyDown(); }); key.addEventListener('touchend', (e) => { handleKeyUp(); }); // Event Listener für Maus (Computer) key.addEventListener('mousedown', (e) => { e.preventDefault(); handleKeyDown(); }); key.addEventListener('mouseup', (e) => { handleKeyUp(); }); }); // Funktion zur Aktualisierung der Overlay-Farbe function updateOverlayColor() { const colors = Array.from(activeKeys).map(keyId => getComputedStyle(document.getElementById(keyId)).backgroundColor); overlay.style.backgroundColor = blendColors(colors); overlay.style.opacity = overlayOpacity; overlay.style.display = 'block'; } // Funktion zum Mischen von Farben function blendColors(colors) { const totalColors = colors.length; const rgbSum = colors.reduce((acc, c) => { const rgb = c.match(/\d+/g).map(Number); return acc.map((val, index) => val + rgb[index]); }, [0, 0, 0]); const r = Math.round(rgbSum[0] / totalColors); const g = Math.round(rgbSum[1] / totalColors); const b = Math.round(rgbSum[2] / totalColors); return `rgba(${r}, ${g}, ${b}, ${overlayOpacity})`; } // Funktion zur Handhabung der Notenanzeige function handleNoteDisplay(key, duration) { let noteType; if (duration >= noteDurations.whole) { noteType = 'whole'; } else if (duration >= noteDurations.half) { noteType = 'half'; } else if (duration >= noteDurations.quarter) { noteType = 'quarter'; } else if (duration >= noteDurations.eighth) { noteType = 'eighth'; } else if (duration >= noteDurations.sixteenth) { noteType = 'sixteenth'; } else { noteType = 'thirtysecond'; } const noteImage = `images2/${noteImages[noteType]}`; const octave = key.getAttribute('data-note') || 0; // Sicherstellen, dass 'data-note' vorhanden ist const topPosition = calculateNotePosition(octave); let img = document.createElement('img'); img.src = noteImage; img.className = `note-img move-fade`; img.style.top = `${topPosition}px`; img.style.right = '0'; noteDisplay.appendChild(img); setTimeout(() => { img.remove(); }, 4000); }
